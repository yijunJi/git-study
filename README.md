##  git 基本命令
##  git config --global alias.st status  # st 替换status
##  git config --global alias.co checkout # co 替换checkout
##  这样，就可以输入git st，或git co来代替命令全称。

##  也可以直接在git的配置文件（~/.gitconfig）中修改：
##  [user]
##  name = xxx
##  email = xxx@gmail.com
##  [alias]
##  co = checkout
##  br = branch
##  ci = commit
##  st = status
